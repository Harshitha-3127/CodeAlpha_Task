LOGIN AND REGISTRATION SYSTEM

#include <iostream>
#include <fstream>
#include <string>
#include <filesystem>
using namespace std;
namespace fs = std::filesystem;

// Function to check if a username already exists (i.e., file exists)
bool userExists(const string& username) {
    return fs::exists(username + ".txt");
}

// Function to register a new user
void registerUser() {
    string username, password;

    cout << "\n--- User Registration ---\n";
    cout << "Enter username: ";
    cin >> username;

    if (userExists(username)) {
        cout << "âŒ Error: Username already exists. Try a different one.\n";
        return;
    }

    cout << "Enter password: ";
    cin >> password;

    ofstream file(username + ".txt");
    file << username << endl;
    file << password << endl;
    file.close();

    cout << "âœ… Registration successful!\n";
}

// Function to login a user
void loginUser() {
    string username, password, storedUser, storedPass;

    cout << "\n--- User Login ---\n";
    cout << "Enter username: ";
    cin >> username;

    if (!userExists(username)) {
        cout << "âŒ Error: Username does not exist.\n";
        return;
    }

    cout << "Enter password: ";
    cin >> password;

    ifstream file(username + ".txt");
    getline(file, storedUser);
    getline(file, storedPass);

    if (username == storedUser && password == storedPass) {
        cout << "âœ… Login successful! Welcome, " << username << "!\n";
    } else {
        cout << "âŒ Error: Invalid password.\n";
    }
}

// Main menu
int main() {
    int choice;

    do {
        cout << "\n==== Login & Registration System ====\n";
        cout << "1. Register\n";
        cout << "2. Login\n";
        cout << "3. Exit\n";

}
